        function formatMoney(amount) {

            return new Intl.NumberFormat('fr-MG', {

                style: 'currency',

                currency: 'MGA',

                maximumFractionDigits: 0

            }).format(amount).replace('MGA', 'Ar');

        }

        function toggleIncomeMode() {

            const isUnknown = document.getElementById('income-unknown').checked;

            const incomeInput = document.getElementById('household-income');

            const budgetInput = document.getElementById('budget-input');

           

            if (isUnknown) {

                incomeInput.disabled = true;

                incomeInput.value = '';

                budgetInput.classList.remove('hidden');

            } else {

                incomeInput.disabled = false;

                budgetInput.classList.add('hidden');

                document.getElementById('monthly-budget').value = '';

            }

        }

        function setPaymentMode(mode) {

            currentMode = mode;

           

            document.querySelectorAll('.payment-mode-btn').forEach(btn => {

                btn.className = 'payment-mode-btn bg-slate-300 text-slate-700 py-2 px-3 rounded-lg text-xs font-bold transition-all';

            });

           

            const activeBtn = document.getElementById(`btn-${mode}`);

            activeBtn.className = 'payment-mode-btn bg-blue-600 text-white py-2 px-3 rounded-lg text-xs font-bold transition-all';

           

            const rateContainer = document.getElementById('rate-container');

            const rentContainer = document.getElementById('rent-container');

           

            if (mode === 'location') {

                rateContainer.classList.add('hidden');

                rentContainer.classList.remove('hidden');

            } else if (mode === 'credit') {

                rateContainer.classList.remove('hidden');

                rentContainer.classList.add('hidden');

            } else {

                rateContainer.classList.add('hidden');

                rentContainer.classList.add('hidden');

            }

        }

        function calculateDownpayment() {

            const price = parseFloat(document.getElementById('property-price').value) || 0;

            const percent = parseFloat(document.getElementById('downpayment-percent').value) || 0;

            const amount = price * (percent / 100);

            document.getElementById('downpayment-amount').textContent = formatMoney(amount);

        }

        document.getElementById('property-price').addEventListener('input', calculateDownpayment);

        document.getElementById('downpayment-percent').addEventListener('input', calculateDownpayment);

        function toggleTable() {

            const content = document.getElementById('table-content');

            const icon = document.getElementById('table-toggle-icon');

            const isHidden = content.classList.contains('hidden');

           

            if (isHidden) {

                content.classList.remove('hidden');

                icon.style.transform = 'rotate(180deg)';

            } else {

                content.classList.add('hidden');

                icon.style.transform = 'rotate(0deg)';

            }

        }

            document.getElementById('household-income').value = '';

            document.getElementById('dependents').value = '0';

            document.getElementById('contributors').value = '1';

        }

                return;

            }

            if (!isIncomeUnknown && householdIncome <= 0) {

                alert('Veuillez entrer un revenu valide ou sélectionner "Inconnu"');

                return;

            }

            if (isIncomeUnknown && monthlyBudget <= 0) {

                alert('Veuillez entrer un budget mensuel valide');

                return;

            }

            const downpayment = propertyPrice * (downpaymentPercent / 100);

            const loanAmount = propertyPrice - downpayment;

           

            if (isIncomeUnknown) {

                householdIncome = monthlyBudget / STANDARD_EFFORT_RATIO;

            }

            let results = {};

           

            if (currentMode === 'credit') {

                results = calculateBankCredit(householdIncome, loanAmount, interestRate, durationYears, downpayment);

            } else if (currentMode === 'location') {

                results = calculateLocationVente(householdIncome, loanAmount, initialRent, durationYears, downpayment);

            } else if (currentMode === 'vefa') {

                results = calculateVEFA(householdIncome, propertyPrice, downpayment, durationYears);

            } else if (currentMode === 'cash') {

                results = calculateCash(householdIncome, propertyPrice);

            }

            displayResults(results, householdIncome, propertyPrice, downpayment);

        }

        function calculateBankCredit(income, loanAmount, annualRate, years, downpayment) {

            const monthlyRate = annualRate / 12;

            const n = years * 12;

            const monthlyPayment = (loanAmount * monthlyRate * Math.pow(1 + monthlyRate, n)) /

                                 (Math.pow(1 + monthlyRate, n) - 1);

           

            const effortRatio = (monthlyPayment / income) * 100;

            const canAfford = monthlyPayment <= income * STANDARD_EFFORT_RATIO;

            const maxBorrowable = (income * STANDARD_EFFORT_RATIO * (Math.pow(1 + monthlyRate, n) - 1)) /

                                (monthlyRate * Math.pow(1 + monthlyRate, n));

            let tableRows = [];

            let remainingCapital = loanAmount;

            let totalInterest = 0;

            let totalPaid = downpayment;

            tableRows.push({

                period: 'Signature',

                acquisition: formatMoney(downpayment),

                rent: '-',

                total: formatMoney(downpayment),

                remaining: formatMoney(remainingCapital),

                isInitial: true

            });

            for (let year = 1; year <= years; year++) {

                let yearlyInterest = 0;

                let yearlyCapital = 0;

               

                for (let month = 1; month <= 12; month++) {

                    const interestMonth = remainingCapital * monthlyRate;

                    const capitalMonth = monthlyPayment - interestMonth;

                    yearlyInterest += interestMonth;

                    yearlyCapital += capitalMonth;

                    remainingCapital -= capitalMonth;

                }

               

                totalInterest += yearlyInterest;

                totalPaid += monthlyPayment * 12;

                tableRows.push({

                    period: `Année ${year}`,

                    acquisition: formatMoney(yearlyCapital),

                    rent: formatMoney(yearlyInterest),

                    total: formatMoney(monthlyPayment * 12),

                    remaining: formatMoney(Math.max(0, remainingCapital)),

                    isInitial: false

                });

            }

            return {

                mode: 'credit',

                monthlyPayment: monthlyPayment,

                totalPaid: totalPaid,

                totalInterest: totalInterest,

                effortRatio: effortRatio,

                canAfford: canAfford,

                maxBorrowable: maxBorrowable,

                tableRows: tableRows,

                breakdown: [

                    { label: 'Apport Personnel', value: formatMoney(downpayment) },

                    { label: 'Montant Emprunté', value: formatMoney(loanAmount) },

                    { label: 'Intérêts Totaux', value: formatMoney(totalInterest) },

                    { label: 'Mensualité Fixe', value: formatMoney(monthlyPayment) }

                ]

            };

        }

        function calculateLocationVente(income, loanAmount, initialRent, years, downpayment) {

            const monthlyAmortization = loanAmount / (years * 12);

            const effortRatio = ((monthlyAmortization + initialRent) / income) * 100;

            const canAfford = (monthlyAmortization + initialRent) <= income * STANDARD_EFFORT_RATIO;

           

            let tableRows = [];

            let remainingCapital = loanAmount;

            let totalPaid = downpayment;

            let currentRent = initialRent;

            tableRows.push({

                period: 'Signature',

                acquisition: formatMoney(downpayment),

                rent: '-',

                total: formatMoney(downpayment),

                remaining: formatMoney(remainingCapital),

                isInitial: true

            });

            for (let year = 1; year <= years; year++) {

                const yearlyAmort = monthlyAmortization * 12;

                const yearlyRent = currentRent * 12;

                remainingCapital -= yearlyAmort;

                totalPaid += yearlyAmort + yearlyRent;

                tableRows.push({

                    period: `Année ${year}`,

                    acquisition: formatMoney(monthlyAmortization),

                    rent: formatMoney(currentRent),

                    total: formatMoney((monthlyAmortization + currentRent) * 12),

                    remaining: formatMoney(Math.max(0, remainingCapital)),

                    isInitial: false

                });

                currentRent *= (1 + ANNUAL_INDEXATION);

            }

            return {

                mode: 'location',

                monthlyPayment: monthlyAmortization + initialRent,

                totalPaid: totalPaid,

                effortRatio: effortRatio,

                canAfford: canAfford,

                tableRows: tableRows,

                breakdown: [

                    { label: 'Apport Personnel', value: formatMoney(downpayment) },

                    { label: 'Montant Financé', value: formatMoney(loanAmount) },

                    { label: 'Amortissement Mensuel', value: formatMoney(monthlyAmortization) },

                    { label: 'Loyer Initial (Indexation 7%/an)', value: formatMoney(initialRent) }

                ]

            };

        }

        function calculateVEFA(income, propertyPrice, downpayment, years) {

            const financed = propertyPrice - downpayment;

            const payment1 = financed * 0.40;

            const payment2 = financed * 0.30;

            const payment3 = financed * 0.30;

           

            const effortRatio = (payment1 / (years * 12) / income) * 100;

            const canAfford = payment1 <= income * years * STANDARD_EFFORT_RATIO;

            let tableRows = [];

            tableRows.push({

                period: 'Apport à la Signature',

                acquisition: formatMoney(downpayment),

                rent: '-',

                total: formatMoney(downpayment),

                remaining: formatMoney(financed),

                isInitial: true

            });

            tableRows.push({

                period: `1ère Tranche (40%) - ${Math.floor(years/3)} mois`,

                acquisition: formatMoney(payment1),

                rent: '-',

                total: formatMoney(payment1),

                remaining: formatMoney(financed - payment1),

                isInitial: false

            });

            tableRows.push({

                period: `2e Tranche (30%) - ${Math.floor(2*years/3)} mois`,

                acquisition: formatMoney(payment2),

                rent: '-',

                total: formatMoney(payment2),

                remaining: formatMoney(financed - payment1 - payment2),

                isInitial: false

            });

            tableRows.push({

                period: `3e Tranche (30%) - Livraison`,

                acquisition: formatMoney(payment3),

                rent: '-',

                total: formatMoney(payment3),

                remaining: formatMoney(0),

                isInitial: false

            });

            return {

                mode: 'vefa',

                monthlyPayment: payment1 / (Math.floor(years/3) * 12),

                totalPaid: propertyPrice,

                effortRatio: effortRatio,

                canAfford: canAfford,

                tableRows: tableRows,

                breakdown: [

                    { label: 'Apport à la Signature', value: formatMoney(downpayment) },

                    { label: '1ère Tranche (40%)', value: formatMoney(payment1) },

                    { label: '2e Tranche (30%)', value: formatMoney(payment2) },

                    { label: '3e Tranche (30%)', value: formatMoney(payment3) }

                ]

            };

        }
